\begin{document}

\begin{frame}[fragile]
  \frametitle{Categories}
  \begin{center}

    \pgfmathsetmacro{\foo}{256}
    \pgfmathsetmacro{\zoom}{1}

    \pgfmathdeclarefunction{toX}{2}{%
      \pgfmathparse{
        (\foo/(2*pi))*2^\zoom*((#2*pi/180) + pi)
      }
    }
    \pgfmathdeclarefunction{toY}{2}{%
      \pgfmathparse{
        (\foo/(2*pi))*2^\zoom*(pi - ln(tan(pi/4 + (#1*pi/180)/2)))
      }
    }

    \makeatletter
    \define@key{worldkeys}{lat}{\def\lat{#1}}
    \define@key{worldkeys}{long}{\def\long{#1}}
    \tikzdeclarecoordinatesystem{world}{
      \setkeys{worldkeys}{#1}
      \pgfpoint{toX(\lat, \long)}{toY(\lat, \long)}
    }
    \begin{tikzpicture}[]
      \node[] (seattle) at (world cs:lat=47.608013,long=-122.335167) {\(seattle\)};
      \node[] (manchester) at (world cs:lat=53.483959,long=-2.244644) {\(manchester\)};
      \node[] (pernith) at (world cs:lat=54.649208,long=-2.755508) {\(pernith\)};
    \end{tikzpicture}
  \end{center}
\end{frame}
